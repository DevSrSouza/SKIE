import useIsBrowser from "@docusaurus/useIsBrowser";
import CodeBlock from "@theme/CodeBlock";
import {interpolate} from "@docusaurus/Interpolate";

export function ApiKeyInterpolatedCodeBlock({children, language, title}) {
    let apiKey = undefined;
    if (useIsBrowser()) {
        const queryString = window.location.search;
        const urlParams = new URLSearchParams(queryString);
        apiKey = urlParams.get('api_key')
    }
    return (
        <CodeBlock language={language} title={title}>
            {interpolate(children, {API_KEY: apiKey || "YOUR_API_KEY_HERE"})}
        </CodeBlock>
    );
}

# Installation

First, you need to choose a version of SKIE that is compatible with your project (see the table below).

| SKIE                     | Kotlin         | Gradle |
|--------------------------|----------------|--------|
| {{SKIE_VERSION}}         | 1.8.0, 1.8.10  | >= 7.3 |
| {{SKIE_VERSION_PREVIEW}} | 1.8.20, 1.8.21 | >= 7.3 |

If your project uses older versions of Kotlin or Gradle, then you will need to upgrade them before you can use SKIE.

:::caution

Before you continue, check that the project compiles (especially if you had to change Gradle and/or Kotlin versions).

:::

SKIE is currently hosted in a private Maven repository, so you will need to register this repository in your Gradle configuration (both for Gradle plugins and for dependencies resolution).
There are multiple ways to do so - depending on your project setup.

One way is to add the following code to `settings.gradle.kts` file:

```kotlin title=settings.gradle.kts
pluginManagement {
    repositories {
        gradlePluginPortal()
        mavenCentral()
        // Add only if you need it for other plugins
        // google()
        maven("https://api.touchlab.dev/public") // <-- Add this
    }
}

// If you don't have a dependencyResolutionManagement block yet, it might be better to add the `repositories` block directly to the `build.gradle.kts` of the module that builds the native framework.
dependencyResolutionManagement {
    repositories {
        mavenCentral()
        // Add only if you need it for other plugins
        // google()
        maven("https://api.touchlab.dev/public") // <-- Add this
    }
}
```

Next, add your API key to `gradle.properties`:

<ApiKeyInterpolatedCodeBlock
    language="properties"
    title="gradle.properties">
    {`touchlab.key={API_KEY}`}
</ApiKeyInterpolatedCodeBlock>

:::info

If you are a trial user, your API key can be found in the trial invitation email.

If you can't find your API key or want to request a trial, [reach out](https://touchlab.co/contact-us/).

:::

Finally, to enable SKIE, add the SKIE plugin to `build.gradle.kts` of the module that builds the native framework:

```kotlin title=build.gradle.kts
plugins {
    // For Kotlin 1.8.0 and 1.8.10
    id("co.touchlab.skie") version "{{SKIE_VERSION}}"

    // For Kotlin 1.8.20 and 1.8.21
    id("co.touchlab.skie") version "{{SKIE_VERSION_PREVIEW}}"
}
```

:::info

SKIE should only be applied to modules that export your Xcode Frameworks. It automatically works on dependent project modules and 3rd party dependencies.

:::

That is the totality of what's required to integrate SKIE!
You can read more about SKIE, and the features it supports, in [SKIE Intro](/docs/intro.md).
If you find that you want or need to customize SKIE's behavior, then check out the [Configuration doc](/docs/Configuration/Configuration.md).
